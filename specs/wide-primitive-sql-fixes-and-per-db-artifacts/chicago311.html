BURSTINESS(area_backlog.date)

(SELECT (STDDEV(gap) - AVG(gap)) / NULLIF(STDDEV(gap) + AVG(gap), 0) FROM (SELECT (EXTRACT(EPOCH FROM sub.date) - EXTRACT(EPOCH FROM LAG(sub.date) OVER (ORDER BY sub.date))) / 86400.0 as gap FROM area_backlog_transform sub WHERE sub.community_area = area_backlog_transform.community_area and sub.date <= aod.as_of_date) gaps WHERE gap IS NOT NULL)
